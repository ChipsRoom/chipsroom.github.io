const createUserBtn = document.getElementById('create-user-btn'); const kickUserInput = document.getElementById('kick-user-input'); const kickUserBtn = document.getElementById('kick-user-btn'); const forceMessageUsername = document.getElementById('force-message-username'); const forceMessageText = document.getElementById('force-message-text'); const forceMessageBtn = document.getElementById('force-message-btn'); const deleteMessageUsername = document.getElementById('delete-message-username'); const deleteMessageText = document.getElementById('delete-message-text'); const deleteMessageBtn = document.getElementById('delete-message-btn'); const deleteAllMessagesBtn = document.getElementById('delete-all-messages-btn'); const searchMessageInput = document.getElementById('search-message-input'); const searchMessageBtn = document.getElementById('search-message-btn'); const onlineUsers = document.getElementById('online-users'); const messageHistory = document.getElementById('message-history'); const searchResults = document.getElementById('search-results'); const roleUsername = document.getElementById('role-username'); const roleSelect = document.getElementById('role-select'); const assignRoleBtn = document.getElementById('assign-role-btn'); let kickedUsers = {}; // Store kicked users let chatHistory = []; // Store messages for later reference // Create User createUserBtn.addEventListener('click', () => { const username = createUserInput.value.trim(); if (username) { const userRef = database.ref('onlineUsers').push(); userRef.set({ username: username }); alert(`${username} has been created.`); createUserInput.value = ''; } else { alert("Please enter a valid username."); } }); // Load Online Users function loadOnlineUsers() { database.ref('onlineUsers').on('value', (snapshot) => { const users = snapshot.val(); onlineUsers.innerHTML = ''; // Clear previous users for (let key in users) { const user = users[key]; if (!kickedUsers[user.username]) { // Don't display kicked users const userElement = document.createElement('div'); userElement.classList.add('user-item'); userElement.innerHTML = `${user.username}`; onlineUsers.appendChild(userElement); } } }); } // Load Message History function loadMessageHistory() { database.ref('chat').on('value', (snapshot) => { const messages = snapshot.val(); messageHistory.innerHTML = ''; // Clear previous messages chatHistory = []; // Reset chat history for (let key in messages) { const message = messages[key]; chatHistory.push(message); // Store messages for searching const messageElement = document.createElement('div'); messageElement.classList.add('message-item'); messageElement.innerHTML = `${message.username}: ${message.message}`; messageHistory.appendChild(messageElement); } }); } // Kick User kickUserBtn.addEventListener('click', () => { const username = kickUserInput.value.trim(); if (username) { database.ref('onlineUsers').once('value', (snapshot) => { const users = snapshot.val(); let userFound = false; // Flag to track if user is found for (let key in users) { if (users[key].username === username) { database.ref('onlineUsers/' + key).remove(); // Remove user kickedUsers[username] = true; // Mark user as kicked alert(`${username} has been kicked.`); userFound = true; // Update flag break; } } if (!userFound) { alert(`User ${username} not found.`); } }); } else { alert("Please enter a valid username."); } }); // Assign User Role assignRoleBtn.addEventListener('click', () => { const username = roleUsername.value.trim(); const role = roleSelect.value; if (username) { database.ref('onlineUsers').once('value', (snapshot) => { const users = snapshot.val(); let userFound = false; // Flag to track if user is found for (let key in users) { if (users[key].username === username) { database.ref('onlineUsers/' + key).update({ role: role }); // Update user role alert(`Role ${role} has been assigned to ${username}.`); userFound = true; // Update flag break; } } if (!userFound) { alert(`User ${username} not found.`); } }); } else { alert("Please enter a valid username."); } }); // Make User Send Message forceMessageBtn.addEventListener('click', () => { const username = forceMessageUsername.value.trim(); const message = forceMessageText.value.trim(); if (username && message) { database.ref('onlineUsers').once('value', (snapshot) => { const users = snapshot.val(); let userFound = false; // Flag to track if user is found for (let key in users) { if (users[key].username === username) { const chatItemRef = database.ref('chat').push(); chatItemRef.set({ username: username, message: message, timestamp: new Date().toISOString() }); alert(`${username} sent the message: "${message}"`); userFound = true; // Update flag break; } } if (!userFound) { alert(`User ${username} not found.`); } }); } else { alert("Please enter both a username and a message."); } }); // Delete Specific Message deleteMessageBtn.addEventListener('click', () => { const username = deleteMessageUsername.value.trim(); const messageToDelete = deleteMessageText.value.trim(); if (username && messageToDelete) { const messageRef = database.ref('chat'); messageRef.once('value', (snapshot) => { const messages = snapshot.val(); let messageFound = false; // Flag to track if message is found for (let key in messages) { const message = messages[key]; if (message.username === username && message.message === messageToDelete) { messageRef.child(key).remove(); // Remove message alert(`Message from ${username} has been deleted.`); messageFound = true; // Update flag break; } } if (!messageFound) { alert(`Message not found for ${username}.`); } }); } else { alert("Please enter both a username and the message text."); } }); // Search Messages searchMessageBtn.addEventListener('click', () => { const keyword = searchMessageInput.value.trim().toLowerCase(); searchResults.innerHTML = ''; // Clear previous results if (keyword) { const filteredMessages = chatHistory.filter(msg => msg.username.toLowerCase().includes(keyword) || msg.message.toLowerCase().includes(keyword) ); if (filteredMessages.length > 0) { filteredMessages.forEach(msg => { const resultElement = document.createElement('div'); resultElement.classList.add('message-item'); resultElement.innerHTML = `${msg.username}: ${msg.message}`; searchResults.appendChild(resultElement); }); } else { searchResults.innerHTML = `
No messages found.
`; } } else { alert("Please enter a keyword to search."); } }); // Delete All Messages deleteAllMessagesBtn.addEventListener('click', () => { if (confirm("Are you sure you want to delete all messages? This action cannot be undone.")) { database.ref('chat').remove(); alert("All messages have been deleted."); messageHistory.innerHTML = ''; // Clear message history display } }); // Load online users and message history on page load loadOnlineUsers(); loadMessageHistory();
